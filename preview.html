7/**
8 * Core game engine loop with fixed timestep physics.
9 */
10public class GameEngine {
11
12 private static final double TICK_RATE = 1.0 / 60.0;
13 private final List<Entity> entities = new ArrayList<>();
14 private boolean running;
15 private double accumulator = 0.0;
16
17 public void start() {
18 running = true;
19 long lastTime = System.nanoTime();
20
21 while (running) {
22 long now = System.nanoTime();
23 double delta = (now - lastTime) / 1_000_000_000.0;
24 lastTime = now;
25 accumulator += delta;
26
27 // Fixed timestep physics updates
28 while (accumulator >= TICK_RATE) {
29 updatePhysics(TICK_RATE);
30 accumulator -= TICK_RATE;
31 }
32
33 render(accumulator / TICK_RATE);
34 }
35 }
36
37 private void updatePhysics(double dt) {
38 for (Entity entity : entities) {
39 Vector3 velocity = entity.getVelocity();
40 Vector3 gravity = Vector3.UP.scale(-9.81);
41
42 velocity = velocity.add(gravity.scale(dt));
43 entity.setVelocity(velocity);
44
45 Vector3 pos = entity.getPosition();
46 entity.setPosition(pos.add(velocity.scale(dt)));
47 }
48 }
49
50 public void addEntity(Entity entity) {
51 if (entity == null) {
52 throw new IllegalArgumentException("Entity cannot be null");
53 }
54 entities.add(entity);

57 private void render(double interpolation) {
58 for (Entity entity : entities) {
